// ====== B: TRẢ LỜI KHI ANH NHẮN (webhook) ======
function doPost(e){
  try{
    var u = JSON.parse(e.postData.contents);
    var m = u.message;
    if(!m || !m.text) return ok_();

    // CHỐNG TRẢ LỜI TRÙNG: mỗi update_id chỉ xử lý 1 lần trong 6 tiếng.
    var cache = CacheService.getScriptCache();
    var key = 'upd_' + u.update_id;
    var lock = LockService.getScriptLock();
    if(!lock.tryLock(15000)) return ok_();
    try{
      if(cache.get(key)) return ok_();
      cache.put(key, '1', 21600);
    } finally { lock.releaseLock(); }

    var chatId = String(m.chat.id);
    if(chatId !== OWNER_CHAT_ID){ tg_(chatId,'Dạ em là thư ký riêng nên chỉ hỗ trợ chủ tiệm thôi ạ.'); return ok_(); }
    var text = m.text.trim();
    var lenh = text.toLowerCase();

    if(lenh === '/start'){
      tg_(chatId,'Dạ em chào anh Tân! Em là trợ lý của anh đây ạ 💕 ' +
        'Anh cần gì cứ nhắn em nha: ý tưởng content, soạn bài đăng, nhắc việc, tư vấn bán hàng...\n\n' +
        'Lệnh nhanh: /tat = ngưng tự nhắn · /bat = nhắn lại · /trangthai = xem bật hay tắt');
      return ok_();
    }
    if(lenh === '/tat' || lenh === '/off'){
      PropertiesService.getScriptProperties().setProperty('TAT_NHAN_CHU_DONG','1');
      tg_(chatId,'Dạ rồi, em ngưng tự nhắn nha anh. Anh nhắn em lúc nào em vẫn trả lời ạ 🤍 (mở lại: /bat)');
      return ok_();
    }
    if(lenh === '/bat' || lenh === '/on'){
      PropertiesService.getScriptProperties().deleteProperty('TAT_NHAN_CHU_DONG');
      tg_(chatId,'Dạ em bật lại tin nhắn mỗi ngày rồi anh nha ☀️');
      return ok_();
    }
    if(lenh === '/trangthai'){
      var tat = PropertiesService.getScriptProperties().getProperty('TAT_NHAN_CHU_DONG');
      var sl = ScriptApp.getProjectTriggers().filter(function(t){
        return t.getHandlerFunction() === 'sendDaily'; }).length;
      tg_(chatId, (tat ? 'Đang TẮT tin nhắn chủ động ạ.' : 'Đang BẬT tin nhắn chủ động ạ.') +
        ' Số lịch nhắn đang chạy: ' + sl + '.');
      return ok_();
    }

    tg_(chatId, gemini_(text));
  }catch(err){}
  return ok_();
}

// ====== A: CHỦ ĐỘNG NHẮN MỖI NGÀY (có khoá chống trùng) ======
function sendDaily(){
  if(PropertiesService.getScriptProperties().getProperty('TAT_NHAN_CHU_DONG')) return;

  var now = new Date();
  var moc = Utilities.formatDate(now, 'Asia/Ho_Chi_Minh', 'yyyyMMdd-HH');
  var props = PropertiesService.getScriptProperties();
  if(props.getProperty('DA_GUI_' + moc)) return;
  props.setProperty('DA_GUI_' + moc, '1');
  donDepMoc_(props);

  var h = Number(Utilities.formatDate(now, 'Asia/Ho_Chi_Minh', 'H'));
  var goi;
  if(h < 10){ goi = 'chào buổi sáng, động viên anh bắt đầu ngày làm việc, gợi ý 1 việc nên làm hôm nay (quay video, đăng bài, nhắn khách cũ...)'; }
  else if(h < 15){ goi = 'hỏi thăm buổi trưa, nhắc anh nghỉ ngơi ăn uống, điểm nhanh 1 việc nên ưu tiên buổi chiều'; }
  else { goi = 'tổng kết nhẹ nhàng cuối ngày, ghi nhận nỗ lực của anh, gợi ý 1 việc chuẩn bị cho ngày mai'; }
  var prompt = 'Viết tin nhắn ngắn (3-4 câu) cho anh Tân, chủ tiệm PPF. Nội dung: ' + goi + '. Ấm áp, tích cực, có 1-2 emoji.';
  tg_(OWNER_CHAT_ID, gemini_(prompt));
}

function donDepMoc_(props){
  var gioiHan = Utilities.formatDate(new Date(Date.now() - 2*86400000), 'Asia/Ho_Chi_Minh', 'yyyyMMdd');
  Object.keys(props.getProperties()).forEach(function(k){
    if(k.indexOf('DA_GUI_') === 0 && k.substring(7,15) < gioiHan) props.deleteProperty(k);
  });
}

// ====== DỌN LỊCH: chạy hàm này để bật/sửa lịch nhắn ======
function setupTriggers(){
  ScriptApp.getProjectTriggers().forEach(function(t){ ScriptApp.deleteTrigger(t); });
  GIO_NHAN.forEach(function(hour){
    ScriptApp.newTrigger('sendDaily').timeBased().everyDays(1).atHour(hour).create();
  });
  Logger.log('Đã dọn sạch. Số lịch còn lại: ' + ScriptApp.getProjectTriggers().length);
}

function xoaHetLich(){

// ====== GỌI AI GEMINI ======
function gemini_(userText){
  var url='https://generativelanguage.googleapis.com/v1beta/models/gemini-flash-latest:generateContent?key='+GEMINI_KEY;
  var payload={ system_instruction:{parts:[{text:PERSONA}]},
    contents:[{role:'user',parts:[{text:userText}]}],
    generationConfig:{temperature:0.85,maxOutputTokens:800} };
  var res=UrlFetchApp.fetch(url,{method:'post',contentType:'application/json',
    payload:JSON.stringify(payload),muteHttpExceptions:true});
  try{ return JSON.parse(res.getContentText()).candidates[0].content.parts[0].text; }
  catch(e){ return 'Dạ em đang bận xíu, anh nhắn lại giúp em nha 🙏'; }
}

// ====== GỬI TIN TELEGRAM ======
function tg_(chatId,text){
  UrlFetchApp.fetch('https://api.telegram.org/bot'+BOT_TOKEN+'/sendMessage',
    {method:'post',contentType:'application/json',
     payload:JSON.stringify({chat_id:chatId,text:text}),muteHttpExceptions:true});
}
function ok_(){ return ContentService.createTextOutput('ok'); }
